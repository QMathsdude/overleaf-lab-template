% Tcolour Boxes Design

% 1. Make slightly curved blue box with line on left
\newtcolorbox{leftbox}{ % Box with color on left
  sharp corners,          % Reset all corners
  colframe=UM-blue,       % Border color
  colback=blue!10!white,  % Background color
  leftrule=1mm,           % Left border thickness
  rightrule=0pt,          % Disable right border
  toprule=0pt,            % Disable top border
  bottomrule=0pt,         % Disable bottom border
  left=2mm,               % Left padding
  right=3mm,              % Right padding
  top=1mm,                % Top padding
  bottom=1mm,             % Bottom padding
  rounded corners=east,
  arc=1mm,
  text width=0.95\linewidth,   % Adjust width as needed
}

% 2. Make slightly curved blue box with line on right
\newtcolorbox{rightbox}{ % Box with color on right

  colframe=UM-blue,         
  colback=blue!10!white,
  leftrule=0pt,        
  rightrule=1mm,          
  toprule=0pt,           
  bottomrule=0pt,  
  left=2mm,              
  right=3mm,              
  top=1mm,              
  bottom=1mm,             
  rounded corners=west,
  arc=1mm,        
  text width=0.95\linewidth,  
}
